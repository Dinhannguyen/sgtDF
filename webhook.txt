-- TÌM KIẾM VÀ LỌC SERVER (Dành cho Metatable)
local function FilterAndJump(serverList)
    if isHopping or type(serverList) ~= "table" then return end
    local bestJobId = nil
    for jobId, info in pairs(serverList) do
        if type(info) == "table" then
            local pCount = tonumber(info.Count) or 0
            local pRegion = tostring(info.Region) or ""
            if jobId ~= game.JobId and string.find(pRegion, targetRegion) and (pCount >= targetCountMin and pCount <= targetCountMax) then
                bestJobId = jobId
                break 
            end
        end
    end
    if bestJobId then
        isHopping = true
        print("[HOP] Đã tìm thấy server (" .. targetRegion .. ")! JobId:", bestJobId)
        pcall(function()
            TeleportService:TeleportToPlaceInstance(game.PlaceId, bestJobId, player)
        end)
    end
end

-- HOOK METATABLE
local mt = getrawmetatable(game)
local oldNamecall = mt.__namecall
setreadonly(mt, false)
mt.__namecall = newcclosure(function(self, ...)
    local method = getnamecallmethod()
    local response = oldNamecall(self, ...)
    if not isHopping and (method == "InvokeServer" or method == "FireServer") and tostring(self.Name) == "__ServerBrowser" then
        if type(response) == "table" then
            task.defer(function()
                FilterAndJump(response)
            end)
        end
    end
    return response
end)
setreadonly(mt, true)

-- REQUEST & WEBHOOK FUNCTIONS
local function getRequestFunc()
    if syn and syn.request then return syn.request, "syn.request" end
    if http and http.request then return http.request, "http.request" end
    if request then return request, "request" end
    if http_request then return http_request, "http_request" end
    if fluxus and fluxus.request then return fluxus.request, "fluxus.request" end
    return nil, "none"
end

local requestFunc, requestType = getRequestFunc()

local function sendWebhookWithRequest(url, jsonBody)
    if not requestFunc then return false, "No request function available" end
    local success, result = pcall(function()
        return requestFunc({
            Url = url,
            Method = "POST",
            Headers = {["Content-Type"] = "application/json"},
            Body = jsonBody
        })
    end)
    return success, result
end

local function sendWebhookWithHttp(url, jsonBody)
    local success, result = pcall(function()
        return HttpService:PostAsync(url, jsonBody, Enum.HttpContentType.ApplicationJson, false)
    end)
    return success, result
end

local function sendWebhook(url, data)
    local jsonBody = HttpService:JSONEncode(data)
    if requestFunc then
        local success, result = sendWebhookWithRequest(url, jsonBody)
        if success then return true, "Sent via " .. requestType end
    end
    local success, result = sendWebhookWithHttp(url, jsonBody)
    if success then
        return true, "Sent via HttpService"
    else
        return false, tostring(result)
    end
end

-- LẤY INVENTORY
local function getInventory()
    if not ReplicatedStorage:FindFirstChild("Remotes") then return nil end
    if not ReplicatedStorage.Remotes:FindFirstChild("CommF_") then return nil end
    local ok, res = pcall(function()
        return ReplicatedStorage.Remotes.CommF_:InvokeServer("getInventory")
    end)
    if ok and typeof(res) == "table" then return res else return nil end
end

local function countDarkFragments(inv)
    if not inv then return 0 end
    local cnt = 0
    for _, item in ipairs(inv) do
        local name = item.Name or item.name or ""
        if name and string.lower(tostring(name)) == "dark fragment" then
            local c = item.Count or item.count or item.Amount or item.amount or item.Quantity or item.quantity or 1
            cnt = cnt + (tonumber(c) or 1)
        end
    end
    return cnt
end

-- HÀM SERVER HOP MỚI (DÙNG UI)
local function serverHopUI()
    if isHopping then return end
    print("[HOP] Bắt đầu khởi động Server Browser...")
    notify("Server Hop", "Đang tìm server " .. targetRegion .. "...", 5)

    local ui = player:WaitForChild("PlayerGui"):WaitForChild("ServerBrowser", 5)
    if ui then
        ui.Enabled = true
        local frame = ui:FindFirstChild("Frame")
        if frame then 
            frame.Visible = true 
            local filters = frame:FindFirstChild("Filters")
            if filters and filters:FindFirstChild("SearchRegion") and filters.SearchRegion:FindFirstChild("TextBox") then
                filters.SearchRegion.TextBox.Text = targetRegion
            end
        end
    end
    
    local startTime = tick()
    task.spawn(function()
        while not isHopping do
            pcall(function()
                local remote = ReplicatedStorage:FindFirstChild("__ServerBrowser")
                if remote then remote:InvokeServer(1, targetRegion) end
            end)
            
            task.wait(3)
            
            if not isHopping then
                if (tick() - startTime) >= 10 then
                    pcall(function()
                        if ui then
                            ui.Enabled = false
                            task.wait(0.5)
                            ui.Enabled = true
                            if ui:FindFirstChild("Frame") then ui.Frame.Visible = true end
                            local remote = ReplicatedStorage:FindFirstChild("__ServerBrowser")
                            if remote then remote:InvokeServer(1, targetRegion) end
                        end
                    end)
                    startTime = tick()
                else
                    if UserInputService.WindowFocused then
                        pcall(function()
                            local gui = player.PlayerGui:FindFirstChild("ServerBrowser")
                            if not gui then return end
                            local frame = gui:FindFirstChild("Frame")
                            if not frame then return end
                            local refreshBtn = frame:FindFirstChild("Refresh")
                            if refreshBtn and refreshBtn.Visible then
                                local x = refreshBtn.AbsolutePosition.X + (refreshBtn.AbsoluteSize.X / 2)
                                local y = refreshBtn.AbsolutePosition.Y + (refreshBtn.AbsoluteSize.Y / 2) + 58
                                VirtualInputManager:SendMouseButtonEvent(x, y, 0, true, game, 1)
                                task.wait(0.05)
                                VirtualInputManager:SendMouseButtonEvent(x, y, 0, false, game, 1)
                            end
                        end)
                    end
                end
            end
        end
    end)
end

    notify("Khởi tạo", "Dark Fragment: " .. lastCount, 4)

        task.wait(CHECK_INTERVAL)
        local currentInv = getInventory()
        if not currentInv then
            warn("[MONITOR] Không lấy được inventory, skip check này")
            continue
        end
        
        local currentCount = countDarkFragments(currentInv)
        
        if currentCount > lastCount then

            notify("Dark Fragment", "Tăng +" .. delta .. " (Tổng: " .. currentCount .. ")", 5)
            
            local webhookData = {
                username = "Dark Fragment Bot",
                avatar_url = "https://cdn.discordapp.com/attachments/1433496764425703424/1434907067662991474/FB_IMG_1760762564286.jpg",
                embeds = {{
                    title = "💠 Dark Fragment Increased!",
                    description = string.format("**%s** nhận được Dark Fragment", player.Name),
                    color = 3066993,
                    fields = {
                        {name = "👤 Player", value = tostring(player.Name), inline = true},
                        {name = "🆔 User ID", value = tostring(player.UserId or 0), inline = true},
                        {name = "💎 Trước", value = tostring(lastCount), inline = true},
                        {name = "💎 Hiện tại", value = tostring(currentCount), inline = true},
                        {name = "➕ Tăng", value = string.format("+%d", delta), inline = true}
                    },
                    footer = {text = "Auto Server Hop in 5s..."},
                    timestamp = os.date("!%Y-%m-%dT%H:%M:%SZ")
                }}
            }
            
            local success, result = sendWebhook(WEBHOOK_URL, webhookData)
            if success then
                print(string.format("✅ [WEBHOOK SENT] +%d Dark Fragment (Total: %d)", delta, currentCount))
                notify("Webhook", "Đã gửi thông tin Dark Fragment!", 4)
            else
                warn("[WEBHOOK ERROR]", result)
                notify("Webhook", "Gửi webhook thất bại!", 4)
            end
            
            for i = COUNTDOWN_BEFORE_HOP, 1, -1 do
                print(string.format("⏳ Hop server trong: %d giây", i))
                notify("Server Hop", "Chuyển server sau " .. i .. " giây", 1)
                task.wait(1)
            end
            
            -- GỌI HÀM HOP MỚI
            print("🔄 Bắt đầu tìm kiếm server mới qua UI...")
            serverHopUI()
            
            local waitCount = 0
            while not isHopping and waitCount < MAX_HOP_TIMEOUT do
                task.wait(1)
                waitCount = waitCount + 1
            end
            
            if isHopping then
                notify("Server Hop", "Đang tiến hành Teleport...", 5)
                task.wait(15) -- Chờ thời gian để quá trình Teleport hoàn tất
            else
                warn(string.format("❌ HOP THẤT BẠI! Đã chờ %d giây mà không tìm được server. Đang kick player...", MAX_HOP_TIMEOUT))
                notify("Server Hop", "Chuyển server timeout! Đang kick...", 6)
                task.wait(2)
                player:Kick("❌ Server Hop thất bại do không tìm được server phù hợp sau " .. MAX_HOP_TIMEOUT .. "s. Vui lòng rejoin!")
            end
            
            lastCount = currentCount
            
        elseif currentCount < lastCount then
            print(string.format("[MONITOR] Dark Fragment giảm: %d -> %d", lastCount, currentCount))
            notify("Dark Fragment", "Giảm: " .. lastCount .. " -> " .. currentCount, 4)
            lastCount = currentCount
        end
    end
end
